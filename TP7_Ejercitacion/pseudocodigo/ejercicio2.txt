Algoritmo busquedaSecuencial(elementos,cantElementos,numeroBuscado)
Inicio
	Para i<cantElementos
		Si elementos[i]==numeroBuscado
			Devolver i
		FinSi
		i++
	FinPara
	Devolver -1
Fin

Algoritmo busquedaBinaria(elementos,cantElementos,numeroBuscado)
Inicio
	sup=cantElementos-1
	Mientras inf<=sup
		centro = ((sup-inf)/2)+inf
		Si elementos[centro]==numeroBuscado
			Devolver centro
		SiNo
			Si numeroBuscado<elementos[centro]
				sup=centro-1
			SiNo
				inf=centro+1
			FinSi
		FinSi
	FinMientras
Fin
